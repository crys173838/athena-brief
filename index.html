async function callAPI(prompt, maxTokens=1200){

  const res=await fetch('https://api.anthropic.com/v1/messages',{

    method:'POST',

    headers:{'Content-Type':'application/json'},

    body:JSON.stringify({

      model:'claude-sonnet-4-20250514',

      max_tokens:Math.min(maxTokens,900),

      // NO tools parameter — browser CORS policy blocks tool_use responses

      // causing sections to return empty and fail silently.

      // News recency is handled via prompt instructions instead.

      messages:[{role:'user',content:prompt}]

    })

  });

  if(!res.ok) throw new Error(`HTTP ${res.status}`);

  const data=await res.json();

  const text=data.content?.filter(b=>b.type==='text')?.map(b=>b.text)?.join('\n')||'';

  if(!text.trim()) throw new Error('Empty response');

  return text;

}

// ── FORMAT ────────────────────────────────────────────────────────────────

function formatContent(raw){

  const parts=raw.split(/TRANSLATION_START|TRANSLATION_END/);

  const main=parts[0]||'';

  const trans=(parts[1]||'').trim();

  let html='<div class="brief-content">'+renderLines(main)+'</div>';

  if(trans){

    const cleanTrans=trans

      .replace(/\[([^\]]+)\]\((https?:\/\/[^\)]+)\)/g,'<a href="$2" target="_blank" rel="noopener">$1</a>')

      .replace(/\*\*(.*?)\*\*/g,'<strong>$1</strong>');

    html+=`<div class="translation-box"><div class="trans-label">🔍 Athena's Translation — Plain English</div><div class="trans-text">${cleanTrans}</div></div>`;

  }

  return html;

}


function renderLines(text){

  text=text.replace(/\[([^\]]+)\]\((https?:\/\/[^\)]+)\)/g,'<a href="$2" target="_blank" rel="noopener">$1</a>');

  let out='';

  text.split('\n').forEach(line=>{

    line=line.trim(); if(!line) return;

    let f=line.replace(/\*\*(.*?)\*\*/g,'<strong>$1</strong>');

    if(line.startsWith('- ')||line.startsWith('• ')) out+=`<ul><li>${f.replace(/^[-•]\s+/,'')}</li></ul>`;

    else if(/^\d+\.\s/.test(line)) out+=`<ul><li>${f.replace(/^\d+\.\s/,'')}</li></ul>`;

    else if(line.toLowerCase().match(/crystal'?s impact|crystal'?s move|bottom line|athena says/)) out+=`<div class="hl-box">${f}</div>`;

    else if(line.toLowerCase().match(/^alert:|^urgent:|^warning:/)) out+=`<div class="al-box">${f}</div>`;

    else if(line.toLowerCase().match(/^action:|^next step:|^today'?s action:/)) out+=`<div class="ac-box">${f}</div>`;

    else out+=`<p>${f}</p>`;

  });

  return out;

}

// ── CHUNKED AUDIO — fixes browser bug where long text silently stops ──────

// Splits full script into sentence-level chunks and queues them sequentially.

// This is the only reliable cross-browser approach for long-form TTS.

let audioChunks=[];

let currentChunkIndex=0;


function splitIntoChunks(text,maxLen=220){

  var parts=text.replace(/([.!?])\s+/g,'$1|').split('|');

  var chunks=[];

  var current='';

  for(var i=0;i<parts.length;i++){

    var s=parts[i];

    if((current+s).length>maxLen&&current){

      chunks.push(current.trim());

      current=s;

    } else {

      current+=(current?' ':'')+s;

    }

  }

  if(current.trim()) chunks.push(current.trim());

  return chunks.filter(function(c){return c.trim().length>3;});

}


function startFrom(pos){

  stopAudio();

  const text=fullAudioScript.slice(pos);

  if(!text.trim()) return;

  if(!selectedVoice) selectBestVoice();


  audioChunks=splitIntoChunks(text);

  currentChunkIndex=0;

  isPlaying=true;

  document.getElementById('btnPlay').textContent='⏸';

  startTracker();

  highlightSec();

  playNextChunk();

}


function playNextChunk(){

  if(!isPlaying||currentChunkIndex>=audioChunks.length){

    if(currentChunkIndex>=audioChunks.length){

      isPlaying=false;

      document.getElementById('btnPlay').textContent='▶';

      stopTracker();

      document.getElementById('audioFill').style.width='100%';

      clearJ();

    }

    return;

  }


  // Mac/Safari fix: must re-select voice each chunk as it can get lost

  if(!selectedVoice) selectBestVoice();


  const chunk=audioChunks[currentChunkIndex];

  const u=new SpeechSynthesisUtterance(chunk);

  u.rate=currentSpeed;

  u.pitch=1.0;

  u.volume=1.0;

  u.lang='en-US';


  // Only set voice if we have one — on Mac, omitting voice uses system default which is fine

  if(selectedVoice) u.voice=selectedVoice;


  u.onstart=()=>{

    // Confirm audio started — helps debug Mac issues

    console.log('Speaking chunk',currentChunkIndex,'/',audioChunks.length);

  };


  u.onend=()=>{

    if(!isPlaying) return;

    currentChunkIndex++;

    const charsSpoken=audioChunks.slice(0,currentChunkIndex).join('').length;

    charProgress=Math.min(charProgress+charsSpoken,fullAudioScript.length);

    highlightSec();

    // Small delay between chunks — helps Mac/Safari not drop utterances

    setTimeout(()=>{ if(isPlaying) playNextChunk(); },80);

  };


  u.onerror=(e)=>{

    console.warn('TTS error on chunk',currentChunkIndex,e.error);

    if(e.error==='interrupted'||e.error==='canceled') return;

    // Skip errored chunk and try next

    currentChunkIndex++;

    setTimeout(()=>{ if(isPlaying) playNextChunk(); },200);

  };


  // Mac Safari fix: cancel any pending speech before speaking new chunk

  // (Safari queues differently from Chrome)

  try{ synth.speak(u); } catch(err){ console.warn('speak error',err); }


  // Chrome/Mac keepalive: Chrome pauses TTS after ~15s in background tabs

  setTimeout(()=>{ if(isPlaying&&synth.speaking) synth.resume(); },10000);

}


function pauseAudio(){

  synth.cancel();

  isPlaying=false;

  document.getElementById('btnPlay').textContent='▶';

  stopTracker();

}

function stopAudio(){

  synth.cancel();

  isPlaying=false;

  audioChunks=[];

  currentChunkIndex=0;

  document.getElementById('btnPlay').textContent='▶';

  stopTracker();

  document.getElementById('audioFill').style.width='0%';

  document.getElementById('audioTime').textContent='0:00 / 0:00';

  clearJ();

}

function clearJ(){document.querySelectorAll('.btn-jump').forEach(b=>b.classList.remove('playing'));document.querySelectorAll('.section-card').forEach(c=>c.classList.remove('playing-now'));}

function jumpTo(key){const o=sectionOffsets.find(s=>s.key===key);if(o){charProgress=o.charStart;startFrom(charProgress);}}

function startTracker(){

  progressStartTime=Date.now(); progressCharAtStart=charProgress; stopTracker();

  audioInterval=setInterval(()=>{

    const el=(Date.now()-progressStartTime)/1000;

    charProgress=Math.min(progressCharAtStart+el*CHARS_PER_SEC*currentSpeed,fullAudioScript.length);

    const pct=charProgress/fullAudioScript.length*100;

    document.getElementById('audioFill').style.width=pct+'%';

    const tot=Math.round(fullAudioScript.length/(CHARS_PER_SEC*currentSpeed));

    const cur=Math.round(charProgress/(CHARS_PER_SEC*currentSpeed));

    document.getElementById('audioTime').textContent=fmtT(cur)+' / '+fmtT(tot);

    highlightSec();

  },300);

}

function stopTracker(){if(audioInterval){clearInterval(audioInterval);audioInterval=null;}}

function fmtT(s){return Math.floor(s/60)+':'+String(s%60).padStart(2,'0');}

function highlightSec(){

  let cur=sectionOffsets[0]?.key;

  for(const s of sectionOffsets){if(charProgress>=s.charStart)cur=s.key;else break;}

  clearJ();

  document.getElementById('jump-'+cur)?.classList.add('playing');

  document.getElementById('card-'+cur)?.classList.add('playing-now');

}

function seekAudio(e){

  const t=document.getElementById('audioTrack'),r=t.getBoundingClientRect();

  const pct=Math.max(0,Math.min(1,(e.clientX-r.left)/r.width));

  charProgress=Math.floor(pct*fullAudioScript.length);

  if(isPlaying) startFrom(charProgress); else document.getElementById('audioFill').style.width=(pct*100)+'%';

}

function setSpeed(s){

  currentSpeed=s;

  document.querySelectorAll('.btn-speed').forEach(b=>b.classList.toggle('active',parseFloat(b.textContent)===s));

  if(isPlaying){pauseAudio();resumeOrStart();}

}
